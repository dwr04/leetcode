class Solution {
public:
    string longestCommonPrefix(vector<string>& strs) {
        int length=strs.size();
        string str=strs[0];
        for(int i=0;i<str.size();++i){
            string s=str.substr(0,i+1);
            for(int j=1;j<length;++j){
                if(strs[j].find(s)!=0){
                    return i==0?"":str.substr(0,i);
                    break;
                }

            }
        }
        return str;
    }
};
