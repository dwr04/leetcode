class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
        int m=nums1.size();
        int n=nums2.size();
        nums1.insert(nums1.end(),nums2.begin(),nums2.end());
        vector<int> nums=nums1;
        sort(nums.begin(),nums.end());
        if((m+n)%2==0){
            return ((nums[(m+n)/2-1]+nums[(m+n)/2])/2.0);
        }
        else{
            return nums[(m+n-1)/2];
        }
    }
};
