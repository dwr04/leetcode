class Solution {
public:
    int maxArea(vector<int>& height) {
        int left=0;
        int right=height.size()-1;
        int maxarea=0;
        int area=0;
        while(left<right){
            area=height[left]>height[right]?(right-left)*height[right]:(right-left)*height[left];
            maxarea=maxarea>area?maxarea:area;
            height[left]>=height[right]?right--:left++;

        }
        return maxarea;
    }
};
